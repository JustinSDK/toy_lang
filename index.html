You can find the source code on Github. Toy lang is the first language I made and might be a prototype if you want to design your own language. The following examples contain several syntaxes of Toy lang. ES6 modules support is required.

  • # Multiplication Table
def print_row(n) {
    i = 2
    while i < 10 {
        print(i + '*' + n + '=' + (i * n) + '\t')
        i = i + 1 
    }
    println()
}

i = 1
while i < 10 {
    print_row(i)
    i = i + 1
}
  • # Tower of Hanoi
def hanoi(n, a, b, c) {
    if n == 1 {
        println('Move sheet from ' + a + ' to ' + c)
    } 
    else {
        hanoi(n - 1, a, c, b)
        hanoi(1, a, b, c)
        hanoi(n - 1, b, a, c) 
    }
}

hanoi(3, 'A', 'B', 'C')
  • # Factorial
def factorial(n) {
    if n == 0 {
        return 1
    }

    return n * factorial(n - 1)
}

n = 1
while n <= 5 {
    println(n + '! = ' + factorial(n))
    n = n + 1
}
  • # Class
class Account {
    # it's a field
    balance = 0 

    def init(number, name) {
        this.number = number
        this.name = name
    }

    def deposit(amount) {
        if amount <= 0 {
            println('must be positive')
            return
        }

        this.balance = this.balance + amount
    }

    def toString() {
        return this.number + ', ' + this.name + ', ' + this.balance
    }
}

acct = new Account('123', 'Justin')
acct.deposit(100)
println(acct)
  • # Built-in Classes
def range(begin, end) {
    lt = new List()
    i = begin
    while i < end {
        lt.append(i)
        i = i + 1
    }
    return lt
}

def sum(lt) {
    if lt.isEmpty() {
        return 0
    }
 
    return lt.get(0) + sum(lt.slice(1))
}
 
lt = range(1, 11)
 
println(lt.join(' + ') + ' = ' + sum(lt))

x = new String('aBc')
println(x.toUpperCase())
println(x.toLowerCase())
