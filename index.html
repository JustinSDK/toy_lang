You can find the source code on Github. Toy lang is the first language I made and might be a prototype if you want to design your own language. The following examples contain several syntaxes of Toy lang. ES6 modules support is required.

  • Multiplication Table
i = 1
while i < 10
    j = 2
    while j < 10
        print j + '*' + i + '=' + (j * i) + '\t'
        j = j + 1
    end
    print '\n'
    i = i + 1
end
  • Fibonacci Numbers
def printFibonacciNumbers(n)
    print n + ' Fibonacci numbers\n'
    n = n - 2
    fn1 = 1
    fn2 = 1
    print fn1 + ' ' + fn2 + ' '
    while n > 0
        fn = fn2 + fn1
        print fn + ' '
        fn1 = fn2
        fn2 = fn
        n = n - 1
    end
    print '\n\n'
end

printFibonacciNumbers(10)
printFibonacciNumbers(20)
  • Tower of Hanoi
def hanoi(n, a, b, c) 
    if n == 1
        print 'Move sheet from ' + a + ' to ' + c + '\n'
    else
        hanoi(n - 1, a, c, b)
        hanoi(1, a, b, c)
        hanoi(n - 1, b, a, c) 
    end
end

hanoi(3, 'A', 'B', 'C')
  • Factorial
def factorial(n)
    if n == 0
        return 1
    end

    return n * factorial(n - 1)
end

n = 1
while n <= 5
    print n + '! = ' + factorial(n) + '\n'
    n = n + 1
end
